# Web App Scaffold Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Scaffold the APARA web app with Bun server + Vite+ React SPA, two-panel layout, and the plumbing for Pi Agent integration.

**Architecture:** Bun serves HTTP + WebSocket on `:3000` (API + static files in prod). Vite+ handles the React SPA dev server on `:5173` with proxy to Bun. The server spawns a Pi Agent subprocess via the existing `PiRpcClient` and bridges it to the browser over WebSocket using a stable app event protocol. shadcn/ui provides the component primitives.

**Tech Stack:** Bun, Vite+ (Vite 8 + Rolldown + Vitest), React, Tailwind CSS, shadcn/ui

**Design Spec:** `doc/specs/2026-04-18-webapp-scaffold-design.md`

```ts
export type ClientMessage =
  | { type: "prompt"; text: string }
  | { type: "abort" }
  | { type: "ping" };

export type ServerMessage =
  | { type: "run_started"; runId: string }
  | { type: "assistant_delta"; runId: string; text: string }
  | { type: "tool_status"; runId: string; tool: string; status: "start" | "end" }
  | { type: "run_finished"; runId: string }
  | { type: "repo_changed" }
  | { type: "error"; code: string; message: string }
  | { type: "pong" };

export function parseClientMessage(raw: string): ClientMessage | null {
  let obj: unknown;
  try {
    obj = JSON.parse(raw);
  } catch {
    return null;
  }

  if (typeof obj !== "object" || obj === null) return null;
  const msg = obj as Record<string, unknown>;

  switch (msg.type) {
    case "prompt":
      if (typeof msg.text !== "string" || msg.text.length === 0) return null;
      return { type: "prompt", text: msg.text };
    case "abort":
      return { type: "abort" };
    case "ping":
      return { type: "ping" };
    default:
      return null;
  }
}

export function parseServerMessage(raw: string): ServerMessage | null {
  let obj: unknown;
  try {
    obj = JSON.parse(raw);
  } catch {
    return null;
  }

  if (typeof obj !== "object" || obj === null) return null;
  const msg = obj as Record<string, unknown>;

  if (typeof msg.type !== "string") return null;

  return msg as unknown as ServerMessage;
}
```

```ts
const COOKIE_NAME = "apara_token";

export function isAuthEnabled(): boolean {
  return typeof process.env.APARA_AUTH_TOKEN === "string" && process.env.APARA_AUTH_TOKEN.length > 0;
}

export function checkAuth(cookieHeader: string | undefined): boolean {
  if (!isAuthEnabled()) return true;

  if (!cookieHeader) return false;

  const token = parseCookie(cookieHeader, COOKIE_NAME);
  return token === process.env.APARA_AUTH_TOKEN;
}

export function createAuthCookie(token: string, useSecure: boolean): string {
  let cookie = `${COOKIE_NAME}=${token}; HttpOnly; SameSite=Strict; Path=/`;
  if (useSecure) {
    cookie += "; Secure";
  }
  return cookie;
}

export function validateOrigin(origin: string | null, expectedHost: string): boolean {
  if (!origin) return false;
  try {
    const url = new URL(origin);
    return url.host === expectedHost;
  } catch {
    return false;
  }
}

function parseCookie(header: string, name: string): string | null {
  const prefix = `${name}=`;
  const parts = header.split(";").map((s) => s.trim());
  for (const part of parts) {
    if (part.startsWith(prefix)) {
      return part.slice(prefix.length);
    }
  }
  return null;
}
```

```ts
import type { PiRpcClient } from "./lib/rpc-client.js";
import type { ServerMessage } from "../src/lib/ws-types.js";
import type { RpcEvent } from "../src/lib/rpc-types.js";
import { randomUUID } from "crypto";

export class PiManager {
  private activeRunId: string | null = null;
  private messageHandler: ((msg: ServerMessage) => void) | null = null;

  constructor(private client: PiRpcClient) {
    this.client.on("event", (event: RpcEvent) => this.mapEvent(event));
    this.client.on("agent_end", () => {
      if (this.activeRunId) {
        this.emit({ type: "run_finished", runId: this.activeRunId });
        this.activeRunId = null;
      }
    });
  }

  handlePrompt(text: string): ServerMessage | undefined {
    if (this.activeRunId) {
      return {
        type: "error",
        code: "busy",
        message: "A prompt is already in progress",
      };
    }

    if (!this.client.isRunning) {
      this.client.start();
    }

    this.activeRunId = randomUUID();
    this.emit({ type: "run_started", runId: this.activeRunId });
    this.client.prompt(text);
    return undefined;
  }

  handleAbort(): void {
    this.client.abort();
  }

  onMessage(handler: (msg: ServerMessage) => void): void {
    this.messageHandler = handler;
  }

  cleanup(): void {
    this.client.stop();
    this.activeRunId = null;
  }

  private emit(msg: ServerMessage): void {
    this.messageHandler?.(msg);
  }

  private mapEvent(event: RpcEvent): void {
    if (!this.activeRunId) return;
    const runId = this.activeRunId;

    switch (event.type) {
      case "message_update": {
        const update = event as {
          type: "message_update";
          assistantMessageEvent: { type: string; delta?: string; [key: string]: unknown };
        };
        if (update.assistantMessageEvent?.type === "text_delta" && typeof update.assistantMessageEvent.delta === "string") {
          this.emit({
            type: "assistant_delta",
            runId,
            text: update.assistantMessageEvent.delta,
          });
        }
        break;
      }
      case "tool_execution_start": {
        const toolEvent = event as { type: "tool_execution_start"; toolName: string };
        this.emit({
          type: "tool_status",
          runId,
          tool: toolEvent.toolName,
          status: "start",
        });
        break;
      }
      case "tool_execution_end": {
        const toolEvent = event as { type: "tool_execution_end"; toolName: string };
        this.emit({
          type: "tool_status",
          runId,
          tool: toolEvent.toolName,
          status: "end",
        });
        break;
      }
    }
  }
}
```

```ts
import { resolve, join } from "path";
import { existsSync, readFileSync, readdirSync, statSync } from "fs";
import { parseArgs } from "util";
import { PiRpcClient } from "./lib/rpc-client.js";
import { PiManager } from "./pi-manager.js";
import { checkAuth, createAuthCookie, isAuthEnabled, validateOrigin } from "./auth.js";
import { parseClientMessage, type ServerMessage } from "../src/lib/ws-types.js";

const { values } = parseArgs({
  args: process.argv.slice(2),
  options: {
    repo: { type: "string" },
    port: { type: "string", default: "3000" },
  },
});

const repoPath = values.repo ?? process.env.APARA_REPO_PATH;
if (!repoPath) {
  console.error(
    "Error: No knowledge repo path provided.\n\n" +
      "Usage:\n" +
      "  bun run server/index.ts --repo /path/to/knowledge-repo\n\n" +
      "Or set the APARA_REPO_PATH environment variable."
  );
  process.exit(1);
}

const resolvedRepo = resolve(repoPath);
const configPath = join(resolvedRepo, ".apara.yaml");
const wikiPath = join(resolvedRepo, "wiki");
const rawPath = join(resolvedRepo, "raw");

if (!existsSync(configPath) && !(existsSync(wikiPath) && existsSync(rawPath))) {
  console.error(
    `Error: ${resolvedRepo} does not appear to be an APARA knowledge repo.\n` +
      "Expected .apara.yaml or wiki/ + raw/ directories."
  );
  process.exit(1);
}

const port = parseInt(values.port ?? "3000", 10);
const hostname = isAuthEnabled() ? "0.0.0.0" : "127.0.0.1";

const distPath = join(import.meta.dir, "..", "dist");
const hasDistFiles = existsSync(distPath);

let piManager: PiManager | null = null;
let activeWs: unknown = null;
let heartbeatTimer: ReturnType<typeof setTimeout> | null = null;

function cleanupSession() {
  piManager?.cleanup();
  piManager = null;
  activeWs = null;
  if (heartbeatTimer) {
    clearTimeout(heartbeatTimer);
    heartbeatTimer = null;
  }
}

function safePath(base: string, requested: string): string | null {
  const full = resolve(base, requested);
  if (!full.startsWith(base)) return null;
  return full;
}

const server = Bun.serve({
  port,
  hostname,

  async fetch(req, server) {
    const url = new URL(req.url);

    // WebSocket upgrade
    if (url.pathname === "/ws") {
      if (isAuthEnabled()) {
        const cookie = req.headers.get("cookie") ?? "";
        if (!checkAuth(cookie)) {
          return new Response("Unauthorized", { status: 401 });
        }
      }
      const upgraded = server.upgrade(req);
      if (!upgraded) {
        return new Response("WebSocket upgrade failed", { status: 400 });
      }
      return undefined;
    }

    // Auth endpoint
    if (url.pathname === "/api/auth" && req.method === "POST") {
      if (!isAuthEnabled()) {
        return Response.json({ ok: true });
      }
      const body = await req.json() as { token?: string };
      if (body.token === process.env.APARA_AUTH_TOKEN) {
        const isSecure = url.protocol === "https:";
        return new Response(JSON.stringify({ ok: true }), {
          headers: {
            "Content-Type": "application/json",
            "Set-Cookie": createAuthCookie(body.token, isSecure),
          },
        });
      }
      // Rate limit: 1s delay on failed attempts
      await new Promise((r) => setTimeout(r, 1000));
      return new Response("Unauthorized", { status: 401 });
    }

    // Protected API routes
    if (url.pathname.startsWith("/api/")) {
      if (isAuthEnabled()) {
        const cookie = req.headers.get("cookie") ?? "";
        if (!checkAuth(cookie)) {
          return new Response("Unauthorized", { status: 401 });
        }
      }

      if (url.pathname === "/api/config") {
        if (existsSync(configPath)) {
          const raw = readFileSync(configPath, "utf-8");
          // Return sanitized — just the key fields, not the raw YAML
          return Response.json({ raw });
        }
        return Response.json({});
      }

      if (url.pathname === "/api/wiki") {
        const requestedPath = url.searchParams.get("path");
        if (requestedPath) {
          const full = safePath(wikiPath, requestedPath);
          if (!full) return new Response("Forbidden", { status: 403 });
          if (!existsSync(full)) return new Response("Not found", { status: 404 });
          return new Response(readFileSync(full, "utf-8"), {
            headers: { "Content-Type": "text/markdown" },
          });
        }
        if (existsSync(wikiPath)) {
          const files = readdirSync(wikiPath, { recursive: true }) as string[];
          return Response.json(files.filter((f) => f.endsWith(".md")));
        }
        return Response.json([]);
      }

      if (url.pathname === "/api/sources") {
        const requestedPath = url.searchParams.get("path");
        if (requestedPath) {
          const full = safePath(rawPath, requestedPath);
          if (!full) return new Response("Forbidden", { status: 403 });
          if (!existsSync(full)) return new Response("Not found", { status: 404 });
          const stat = statSync(full);
          // Text files under 1MB: return content. Otherwise: metadata only.
          if (stat.size < 1_000_000 && requestedPath.match(/\.(md|txt|yaml|yml|json|csv|tsv)$/i)) {
            return new Response(readFileSync(full, "utf-8"), {
              headers: { "Content-Type": "text/plain" },
            });
          }
          return Response.json({ name: requestedPath, size: stat.size, type: "binary" });
        }
        if (existsSync(rawPath)) {
          const files = readdirSync(rawPath, { recursive: true }) as string[];
          return Response.json(
            files.map((f) => {
              const full = join(rawPath, f);
              const stat = statSync(full);
              return { name: f, size: stat.size, isDirectory: stat.isDirectory() };
            })
          );
        }
        return Response.json([]);
      }

      if (url.pathname === "/api/log") {
        const logPath = join(wikiPath, "log.md");
        if (existsSync(logPath)) {
          return new Response(readFileSync(logPath, "utf-8"), {
            headers: { "Content-Type": "text/markdown" },
          });
        }
        return new Response("", { headers: { "Content-Type": "text/markdown" } });
      }

      return new Response("Not found", { status: 404 });
    }

    // Serve static files in production
    if (hasDistFiles) {
      let filePath = join(distPath, url.pathname === "/" ? "index.html" : url.pathname);
      if (existsSync(filePath)) {
        return new Response(Bun.file(filePath));
      }
      // SPA fallback
      return new Response(Bun.file(join(distPath, "index.html")));
    }

    return new Response("Not found", { status: 404 });
  },

  websocket: {
    open(ws) {
      if (activeWs) {
        ws.close(1008, "Only one session at a time");
        return;
      }
      activeWs = ws;

      const client = new PiRpcClient({
        cwd: resolvedRepo,
        extensionPath: resolve(import.meta.dir, "../../extension/apara.ts"),
      });

      piManager = new PiManager(client);
      piManager.onMessage((msg: ServerMessage) => {
        ws.send(JSON.stringify(msg));
      });

      // Reset heartbeat
      heartbeatTimer = setTimeout(() => {
        ws.close(1001, "Heartbeat timeout");
        cleanupSession();
      }, 60_000);
    },

    message(ws, message) {
      const raw = typeof message === "string" ? message : new TextDecoder().decode(message);
      const parsed = parseClientMessage(raw);
      if (!parsed) return;

      // Reset heartbeat on any message
      if (heartbeatTimer) clearTimeout(heartbeatTimer);
      heartbeatTimer = setTimeout(() => {
        ws.close(1001, "Heartbeat timeout");
        cleanupSession();
      }, 60_000);

      switch (parsed.type) {
        case "prompt": {
          const error = piManager?.handlePrompt(parsed.text);
          if (error) ws.send(JSON.stringify(error));
          break;
        }
        case "abort":
          piManager?.handleAbort();
          break;
        case "ping":
          ws.send(JSON.stringify({ type: "pong" }));
          break;
      }
    },

    close() {
      setTimeout(() => cleanupSession(), 5000);
    },
  },
});

console.log(`APARA server running on http://${hostname}:${port}`);
console.log(`Knowledge repo: ${resolvedRepo}`);
if (isAuthEnabled()) {
  console.log("Auth: enabled (APARA_AUTH_TOKEN set)");
} else {
  console.log("Auth: disabled (local mode, bound to 127.0.0.1)");
}

process.on("SIGINT", () => {
  cleanupSession();
  server.stop();
  process.exit(0);
});

process.on("SIGTERM", () => {
  cleanupSession();
  server.stop();
  process.exit(0);
});
```
